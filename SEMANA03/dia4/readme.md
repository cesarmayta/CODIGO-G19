#PASOS PARA DESPLIEGUE

PASO 1 - CREAR ENTORNO VIRTUAL
python -m venv venv

PASO 2 - ACTIVAR ENTORNO VIRTUAL
source venv/Scripts/activate

pip install -r requirements.txt

export FLASK_APP=run.py
export FLASK_DEBUG=1

PASO 5 - DESPLIEGUE
flask run
